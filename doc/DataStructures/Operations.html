def diff(other)
  new_set = Sets.new
  items.each { |item| new_set.add(item) unless other.item_available?(item) }
  new_set
end

  !items.select { |item| other.item_available?(item) }.empty?
end

def union(other)
  new_set = Sets.new
  items.each { |item| new_set.add(item) }
  other.items.each { |item| new_set.add(item) }
  new_set
end
